local function setupSliderLogic(track, fill, btn, box, getVal, setVal, name)
    local minVal, maxVal = 0, 100
    
    local function updateFromPercentage(percentage)
        local rawVal = minVal + (percentage * (maxVal - minVal))
        setVal(math.floor(rawVal * 10) / 10)
        btn.Position = UDim2.new(percentage, -5, -1.25, 0)
        fill.Size = UDim2.new(percentage, 0, 1, 0)
        box.Text = string.format("%.1f", getVal())
    end

    local active = false
    btn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then active = true end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then active = false end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if active and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local percentage = math.clamp((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
            updateFromPercentage(percentage)
        end
    end)
    
    box.FocusLost:Connect(function()
        local val = tonumber(box.Text)
        if val then
            setVal(math.clamp(val, minVal, maxVal))
            local pct = (getVal() - minVal) / (maxVal - minVal)
            btn.Position = UDim2.new(pct, -5, -1.25, 0)
            fill.Size = UDim2.new(pct, 0, 1, 0)
            box.Text = string.format("%.1f", getVal())
        else
            box.Text = string.format("%.1f", getVal())
        end
    end)
    
    local initPct = (getVal() - minVal) / (maxVal - minVal)
    btn.Position = UDim2.new(initPct, -5, -1.25, 0)
    fill.Size = UDim2.new(initPct, 0, 1, 0)
    box.Text = string.format("%.1f", getVal())
end

local dragging, dragInput, dragStart, startPos
MainFrame.InputBegan:Connect(function(input)
    if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) then
        local clickedControls = false
        if input.Position.Y > ControlsContainer.AbsolutePosition.Y and input.Position.Y < (ControlsContainer.AbsolutePosition.Y + ControlsContainer.AbsoluteSize.Y) then
            if input.Position.X > ControlsContainer.AbsolutePosition.X and input.Position.X < (ControlsContainer.AbsolutePosition.X + ControlsContainer.AbsoluteSize.X) then
                clickedControls = true
            end
        end
        if not clickedControls then
            dragging = true
            dragStart = input.Position
            startPos = MainFrame.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end
end)

    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then dragInput = input end
end)

local function refreshballs(force)
    if not force and lastrefreshtime + refreshInterval > os.time() then return end
    lastrefreshtime = os.time()
    table.clear(balls)
    for _, v in pairs(workspace:GetDescendants()) do
        if ballNames[v.Name] then table.insert(balls, v) end
    end
    if isEnabled then
        Log3.Text = "TARGETS FOUND IN WORKSPACE: " .. #balls
    end
end

refreshballs(true)

local function findCurrentBall()
    for _, obj in ipairs(workspace:GetDescendants()) do
        if ballNames[obj.Name] and obj:IsA("BasePart") then
            return obj
        end
    end
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj:IsA("BasePart") and (obj.Shape == Enum.PartType.Ball or obj.Name:lower():find("ball")) then
            return obj
        end
    end
    return nil
end

local function predictLandingZone(ball)
    local position = ball.Position
    local velocity = ball.AssemblyLinearVelocity
    local bv = ball:FindFirstChildWhichIsA("BodyVelocity") or ball:FindFirstChildWhichIsA("LinearVelocity")
    if bv then
        if bv:IsA("BodyVelocity") then
            velocity = bv.Velocity
        elseif bv:IsA("LinearVelocity") then
            velocity = bv.VectorVelocity
        end
    end
    if velocity.Magnitude < 1.5 then
        return nil
    end
    local gravityVec = Vector3.new(0, -workspace.Gravity, 0)
    for step = 1, 180 do
        velocity = velocity + (gravityVec * (1/60))
        local nextPosition = position + (velocity * (1/60))
        local raycastParams = RaycastParams.new()
        raycastParams.FilterType = Enum.RaycastFilterType.Exclude
        raycastParams.FilterDescendantsInstances = {ball, TargetMarker, player.Character}
        local raycastResult = workspace:Raycast(position, nextPosition - position, raycastParams)
        if raycastResult then
            return raycastResult.Position
        end
        position = nextPosition
    end
    return nil
end

RunService.RenderStepped:Connect(function()
    if not isEnabled or not predictorEnabled then 
        TargetMarker.Transparency = 1 
        return 
    end
    local ball = findCurrentBall()
    if ball then
        local landingPosition = predictLandingZone(ball)
        if landingPosition then
            TargetMarker.Transparency = 0.3
            TargetMarker.CFrame = CFrame.new(landingPosition + Vector3.new(0, 0.1, 0)) * CFrame.Angles(0, 0, math.rad(90))
        else
            TargetMarker.Transparency = 1
        end
    else
        TargetMarker.Transparency = 1
    end
end)

local inputConnection
inputConnection = UserInputService.InputBegan:Connect(function(input, gameProcessedEvent)
    if isBindingToggle and input.UserInputType == Enum.UserInputType.Keyboard then
        TOGGLE_KEY = input.KeyCode
        BindToggleBtn.Text = "Toggle Key: " .. input.KeyCode.Name
        BindToggleBtn.TextColor3 = Color3.fromRGB(200, 220, 255)
        isBindingToggle = false
        return
    end

    if isBindingHide and input.UserInputType == Enum.UserInputType.Keyboard then
        HIDE_KEY = input.KeyCode
        BindHideBtn.Text = "Hide UI Key: " .. input.KeyCode.Name
        BindHideBtn.TextColor3 = Color3.fromRGB(200, 220, 255)
        isBindingHide = false
        return
    end

    if input.KeyCode == HIDE_KEY then
        ScreenGui.Enabled = not ScreenGui.Enabled
        if ScreenGui.Enabled then
            showNotification("UI VISIBLE", "Control framework has been restored.", Color3.fromRGB(85, 255, 170))
        else
            showNotification("UI HIDDEN", "System window minimized to taskbar.", Color3.fromRGB(255, 170, 0))
        end
        return
    end

    if input.KeyCode == DESTROY_KEY then
        if inputConnection then
            inputConnection:Disconnect()
            ScreenGui:Destroy()
            NotifGui:Destroy()
            TargetMarker:Destroy()
        end
        return
    end

    if input.KeyCode == TOGGLE_KEY then
        setSystemState(not isEnabled)
        return
    end
    
    if gameProcessedEvent then return end
    
    if input.KeyCode == Enum.KeyCode.W or input.KeyCode == Enum.KeyCode.A or 
       input.KeyCode == Enum.KeyCode.S or input.KeyCode == Enum.KeyCode.D or 
       input.KeyCode == Enum.KeyCode.Space then
        return 
    end

    local character = player.Character
    if not character then return end

    local activeLegs = (toggleLegs and reachLegs > 0)
    local activeArms = (toggleArms and reachArms > 0)
    local activeHead = (toggleHead and reachHead > 0)
    local activeTorso = (toggleTorso and reachTorso > 0)

    local function safeToggleParts(partNames, state)
        for _, name in ipairs(partNames) do
            local p = character:FindFirstChild(name)
            if p and p:IsA("BasePart") then
                p.CanTouch = state
            end
        end
    end

    safeToggleParts({"Right Leg", "Left Leg", "RightUpperLeg", "RightLowerLeg", "LeftUpperLeg", "LeftLowerLeg", "RightFoot", "LeftFoot"}, activeLegs)
    safeToggleParts({"Right Arm", "Left Arm", "RightUpperArm", "RightLowerArm", "LeftUpperArm", "LeftLowerArm", "Right Hand", "Left Hand"}, activeArms)
    safeToggleParts({"Head"}, activeHead)
    safeToggleParts({"Torso", "UpperTorso", "LowerTorso"}, activeTorso)

    if not isEnabled then return end

    refreshballs(false)
    local executionQueues = {}

    if activeLegs then
        local parts = {}
        for _, n in ipairs({"Right Leg", "Left Leg", "RightUpperLeg", "RightLowerLeg", "LeftUpperLeg", "LeftLowerLeg", "RightFoot", "LeftFoot"}) do
            local p = character:FindFirstChild(n) if p then table.insert(parts, p) end
        end
        table.insert(executionQueues, {parts = parts, range = reachLegs})
    end
    
    if activeArms then
        local parts = {}
        for _, n in ipairs({"Right Arm", "Left Arm", "RightUpperArm", "RightLowerArm", "LeftUpperArm", "LeftLowerArm", "Right Hand", "Left Hand"}) do
            local p = character:FindFirstChild(n) if p then table.insert(parts, p) end
        end
        table.insert(executionQueues, {parts = parts, range = reachArms})
    end
    
    if activeHead then
        local p = character:FindFirstChild("Head")
        if p then table.insert(executionQueues, {parts = {p}, range = reachHead}) end
    end
    
    if activeTorso then
        local parts = {}
        for _, n in ipairs({"Torso", "UpperTorso", "LowerTorso"}) do
            local p = character:FindFirstChild(n) if p then table.insert(parts, p) end
        end
        table.insert(executionQueues, {parts = parts, range = reachTorso})
    end

    if #executionQueues == 0 then return end

    local touched = false 

    for _, e in pairs(balls) do
        if e:IsA("BasePart") and e.Parent then
            for _, queue in pairs(executionQueues) do
                for _, part in pairs(queue.parts) do
                    if (e.Position - part.Position).Magnitude < queue.range then
                        for _, v in pairs(part:GetDescendants()) do
                            if v.Name == "TouchInterest" and v.Parent then
                                firetouchinterest(e, v.Parent, 0)
                                touched = true
                                break
                            end
                        end
                    end
                    if touched then break end
                end
                if touched then break end
            end
        end
        if touched then break end
    end
end)
